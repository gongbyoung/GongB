01

전체 구조 이해하기

모든 비주얼라이저는 draw 함수 하나로 이루어져 있습니다. 이 함수는 매 프레임(초당 약 60번) 호출되며, 캔버스에 그림을 그립니다.

기본 구조 
// ① 함수 선언 — 이름만 바꾸면 됩니다
function drawMyViz(ctx, W, H, t, freq) {
  try {

    // ② 중심점 계산
    const cx = W / 2;
    const cy = H / 2;

    // ③ 음악 에너지 추출
    let bass = 0;
    for (let i = 0; i < 10; i++) bass += (freq[i] || 0);
    bass = bass / 10 / 255; // 결과: 0.0 ~ 1.0

    // ④ 실제 그리기
    ctx.beginPath();
    ctx.arc(cx, cy, 50 + bass * 100, 0, Math.PI * 2);
    ctx.strokeStyle = `hsla(${t * 30}, 80%, 60%, 0.8)`;
    ctx.lineWidth = 2;
    ctx.stroke();

  } catch(e) {} // 에러 격리 — 항상 유지
}
파라미터 5가지

ctx — 캔버스 2D 컨텍스트 (그리는 도구)
W — 캔버스 너비 (픽셀)
H — 캔버스 높이 (픽셀)
t — 경과 시간 (계속 증가, 애니메이션용)
freq — 주파수 데이터 배열 (Uint8Array)

배경 지우기

매 프레임 배경을 반투명하게 지우면 잔상 효과가 생깁니다.

ctx.fillStyle = 'rgba(0,0,0,0.2)'
ctx.fillRect(0,0,W,H)

0.2 → 긴 잔상 / 0.5 → 짧은 잔상 / 1.0 → 잔상 없음

try/catch 필수

함수 전체를 try { ... } catch(e) {}로 감싸야 합니다. 하나의 효과에서 에러가 나도 나머지 카드들이 계속 동작합니다.

02

핵심 파라미터 — freq / t / amp

freq 배열 — 주파수 데이터

freq는 0~255 사이 정수값 1024개가 담긴 배열입니다. 인덱스 0에 가까울수록 저음(베이스), 높을수록 고음(트레블)입니다.

범위음역사용 예
freq[0~10]저음 (Bass)킥드럼, 베이스기타 — 화면 전체 흔들기
freq[10~50]중저음리듬감, 박자 — 도형 크기 변화
freq[50~200]중음 (Mid)멜로디, 보컬 — 색상 변화
freq[200~512]고음 (Hi)심벌즈, 노이즈 — 파티클 밀도
에너지 추출 패턴 
// 베이스 에너지 (0~1)
let bass = 0;
for (let i = 0; i < 10; i++) bass += (freq[i] || 0);
bass = bass / 10 / 255;

// 전체 에너지 (0~1)
let energy = 0;
for (let i = 0; i < freq.length; i++) energy += (freq[i] || 0);
energy = energy / freq.length / 255;

// 특정 인덱스의 값 (0~1) — 루프에서 자주 사용
const fi = Math.min(Math.floor((i / total) * freq.length), freq.length - 1);
const amp = (freq[fi] || 0) / 255; // 결과: 0.0 ~ 1.0

t — 시간 변수

매 프레임 약 0.012씩 증가합니다. Math.sin(t)와 함께 쓰면 부드러운 주기 움직임이 됩니다.

t 활용 패턴 
// 회전 — 각도에 t를 더함
const angle = (i / total) * Math.PI * 2 + t;

// 흔들림 — sin/cos에 t 곱하기
const wobble = Math.sin(t * 3) * 20;     // 빠른 진동
const sway   = Math.sin(t * 0.5) * 50;   // 느린 흔들림

// 색상 순환
const hue = (t * 30) % 360;  // 색상환을 천천히 순환
const hue = (t * 120) % 360; // 빠르게 순환

// 점멸 (0~1 사이 왕복)
const blink = (Math.sin(t * 5) + 1) / 2;

기본 도형 그리기

원 (arc)

// 원 하나 그리기
ctx.beginPath();
ctx.arc(x, y, radius, 0, Math.PI * 2);
ctx.strokeStyle = 'rgba(100, 200, 255, 0.8)';
ctx.lineWidth = 2;
ctx.stroke();           // 테두리만
// ctx.fill();          // 채우기만
// ctx.stroke(); ctx.fill(); // 둘 다

원형으로 점 배치하기 — 가장 많이 쓰는 패턴

const total = 64;           // 점의 수
const radius = 150;          // 원의 반지름

for (let i = 0; i < total; i++) {

  // ① i를 각도로 변환
  const angle = (i / total) * Math.PI * 2;

  // ② freq에서 이 각도의 음량 가져오기
  const fi  = Math.min(Math.floor((i / total) * freq.length), freq.length - 1);
  const amp = (freq[fi] || 0) / 255;   // 0~1

  // ③ 반지름을 amp로 늘이기
  const r = radius + amp * 80;

  // ④ 극좌표 → 직교좌표 변환
  const x = cx + Math.cos(angle) * r;
  const y = cy + Math.sin(angle) * r;

  // ⑤ 점 그리기
  ctx.beginPath();
  ctx.arc(x, y, 3 + amp * 6, 0, Math.PI * 2);
  ctx.fillStyle = `hsla(${i * 5 + t * 30}, 80%, 65%, ${0.5 + amp * 0.5})`;
  ctx.fill();
}

폴리라인 (연결선으로 파형 그리기)

// 원형 파형 — beginPath 한 번으로 전체 연결
ctx.beginPath();
for (let i = 0; i <= 128; i++) {
  const angle = (i / 128) * Math.PI * 2;
  const fi  = Math.min(Math.floor((i / 128) * freq.length), freq.length - 1);
  const amp = (freq[fi] || 0) / 255;
  const r = 120 + amp * 80;
  const x = cx + Math.cos(angle) * r;
  const y = cy + Math.sin(angle) * r;
  i === 0 ? ctx.moveTo(x, y) : ctx.lineTo(x, y);
}
ctx.closePath(); // 마지막 점과 첫 점을 연결
ctx.strokeStyle = `hsla(${t * 20}, 80%, 60%, 0.8)`;
ctx.lineWidth = 2;
ctx.stroke();

그라디언트 (Radial Gradient — 빛나는 중심)

// 중심에서 바깥으로 퍼지는 빛
const grd = ctx.createRadialGradient(
  cx, cy, 0,                     // 안쪽 원: 중심, 반지름 0
  cx, cy, Math.max(1, 80 + bass * 60)  // 바깥 원: 중심, 반지름 (Math.max(1,...) 필수!)
);
grd.addColorStop(0,   'rgba(255, 200, 100, 0.9)'); // 중심색
grd.addColorStop(0.5, 'rgba(255, 80, 20, 0.3)');  // 중간색
grd.addColorStop(1,   'rgba(255, 0, 0, 0)');      // 투명 가장자리

ctx.beginPath();
ctx.arc(cx, cy, Math.max(1, 80 + bass * 60), 0, Math.PI * 2);
ctx.fillStyle = grd;
ctx.fill();

⚠️ 주의: createRadialGradient의 마지막 인자(반지름)가 0이 되면 에러가 납니다. 반드시 Math.max(1, ...)로 최솟값을 보장하세요.

04

음악 반응 공식 모음

아래 공식들을 복사해서 숫자만 바꾸면 다른 느낌이 납니다.

크기 반응

radius = 기본크기 + amp × 최대증가량
radius = 100 + bass × 80 // 베이스에 반응하는 원
size = 3 + amp × 10 // 파티클 크기

회전 반응

angle = (i / total) × 2π + t × 속도
angle = (i / total) × 2π + t × 0.5 + energy × 2 // 소리 크면 빠르게 회전

파동 반응

y = Math.sin(x × 주파수 + t × 속도) × 진폭 × amp
distort = Math.sin(angle × 3 + t × 2) × amp × radius × 0.3

색상 반응

hue = 기본색상 + amp × 색상변화폭
hue = 200 + amp × 120 // 파란→초록→노랑 변화
hue = i × (360 / total) + t × 30 // 무지개 + 시간 회전

투명도 반응

alpha = 0.3 + amp × 0.7 // 소리 없으면 30%, 최대 100%
alpha = (1 - i / total) × (0.3 + amp) // 뒤로 갈수록 흐릿 (트레일 효과)
🎯 베이스 킥 감지

베이스가 임계값을 넘을 때 효과 트리거

if (bass > 0.6) {
  // 킥드럼 순간 효과
  particles.push({...});
}
📐 극좌표 → 직교좌표

원형 배치의 핵심 공식

x = cx + Math.cos(angle) * r;
y = cy + Math.sin(angle) * r;
🌊 sin으로 물결

부드러운 파동 만들기

wave = Math.sin(x * 0.02 + t * 2)
     * amp * 50;
05

색상 — hsla로 살아있는 색 만들기

hsla(색조, 채도%, 밝기%, 투명도)는 가장 직관적인 색상 표현입니다.

값범위설명
H (색조)0~3600=빨강, 60=노랑, 120=초록, 180=하늘, 240=파랑, 300=보라
S (채도)0~100%100%=선명, 0%=회색
L (밝기)0~100%50%=표준, 0%=검정, 100%=흰색
A (투명도)0~10=투명, 1=불투명
색상 레시피 
// 음악에 따라 색상환 순환
`hsla(${t * 30 % 360}, 80%, 60%, 0.8)`

// 원형 배치에서 무지개색
`hsla(${i * (360 / total)}, 85%, 65%, ${0.5 + amp * 0.5})`

// 베이스에 따라 빨강 ↔ 파랑
`hsla(${240 - bass * 240}, 90%, 60%, 0.8)`

// 에너지에 따라 차갑게 ↔ 따뜻하게
`hsla(${200 + energy * 120}, 80%, ${40 + energy * 30}%, 0.8)`

// 불꽃 색상 (빨강~노랑)
`hsla(${amp * 60}, 100%, ${50 + amp * 30}%, ${0.4 + amp * 0.6})`

// 레이어마다 색상 이동
`hsla(${baseHue + layer * 20}, 80%, 60%, 0.6)`
06

상태 기억하기 — 파티클 / 트레일

파티클은 함수 바깥에 선언해서 프레임 간 상태를 유지합니다.

파티클 시스템 기본 패턴 
// ① 함수 밖에서 파티클 배열 선언 (한 번만 초기화)
const _myParticles = Array.from({ length: 100 }, () => ({
  x:     Math.random() * 2 - 1,   // -1 ~ 1 (나중에 W/H 곱함)
  y:     Math.random() * 2 - 1,
  vx:    (Math.random() - 0.5) * 0.01,  // 속도
  vy:    (Math.random() - 0.5) * 0.01,
  size:  Math.random() * 3 + 1,
  hue:   Math.random() * 360,
}));

function drawMyParticles(ctx, W, H, t, freq) { try {
  let energy = 0;
  for (let i = 0; i < freq.length; i++) energy += freq[i];
  energy /= freq.length * 255;

  _myParticles.forEach(p => {
    // ② 매 프레임 위치 업데이트
    p.x += p.vx * (1 + energy * 5);
    p.y += p.vy * (1 + energy * 5);

    // ③ 화면 벗어나면 반대편으로
    if (p.x > 1)  p.x = -1;
    if (p.x < -1) p.x =  1;
    if (p.y > 1)  p.y = -1;
    if (p.y < -1) p.y =  1;

    // ④ 실제 픽셀 좌표로 변환해서 그리기
    const px = (p.x + 1) / 2 * W;
    const py = (p.y + 1) / 2 * H;
    ctx.beginPath();
    ctx.arc(px, py, p.size * (1 + energy * 2), 0, Math.PI * 2);
    ctx.fillStyle = `hsla(${p.hue + t * 20}, 80%, 65%, 0.7)`;
    ctx.fill();
  });
} catch(e) {} }

트레일 (꼬리) 패턴

// 과거 위치를 배열로 저장해서 꼬리 만들기
const _trail = [];

// 매 프레임: 현재 위치를 앞에 추가
_trail.unshift({ x: headX, y: headY, alpha: 1 });

// 길이 제한
if (_trail.length > 30) _trail.pop();

// 꼬리 그리기
_trail.forEach((pt, i) => {
  const alpha = (1 - i / _trail.length) * 0.8;
  const size  = (1 - i / _trail.length) * 8;
  ctx.beginPath();
  ctx.arc(pt.x, pt.y, Math.max(0.1, size), 0, Math.PI * 2);
  ctx.fillStyle = `rgba(100, 200, 255, ${alpha})`;
  ctx.fill();
});

09

기존 효과 변형 레시피

숫자 하나만 바꿔도 전혀 다른 효과가 나옵니다.

DNA Helix 변형

strands = 2 → 3 또는 5로 바꾸면 더 복잡한 나선

r*0.4 (타원비율) → r*0.8으로 바꾸면 더 원형에 가까운 궤도

radius*0.15 (흔들림) → radius*0.4로 키우면 격렬하게 흔들림

Vortex Tunnel 변형

RINGS = 28 → 50으로 늘리면 더 깊은 터널

SPOKES = 64 → 128으로 늘리면 더 촘촘한 격자

t * 1.5 (회전속도) → t * 0.3으로 줄이면 천천히 흡입

Nebula 변형

length: 300 (파티클 수) → 800으로 늘리면 더 빽빽한 성운

speed * 0.02 → 0.05로 키우면 빠르게 소용돌이

hue: 200 → 0으로 바꾸면 붉은 성운으로 변환

Electric Web 변형

length: 16 (노드 수) → 24로 늘리면 더 촘촘한 그물

maxD = baseR * 1.4 → * 2.0으로 늘리면 더 멀리 연결

번개 depth 2 → 3으로 늘리면 더 구불구불한 번개

새 효과 조합 아이디어

행성 시스템: Atomic Orbit + 파티클 꼬리
음파 폭발: Supernova + Grid Pulse 합성
바다 폭풍: Ocean Wave + Lightning Storm 결합
생물체: Tentacle + DNA 나선 조합

배경색으로 분위기 전환

VIZS 배열의 bg 값 변경:

'#000000' → 완전한 블랙
'#050010' → 딥 네이비
'#100000' → 다크 레드
'#001008' → 다크 그린

완전히 새로운 효과를 만드는 사고 과정

①
시각적 목표 정하기
"물방울이 퍼지는 효과", "별자리처럼 점들이 연결되는 효과" 등 먼저 말로 표현
②
어떤 음악 요소에 반응할지 결정
크기 → bass, 속도 → energy, 색상 → amp, 개수 → mid 등 매핑
③
위의 라이브 에디터에서 실험
에디터에서 형태를 잡고 → 만족스러우면 music_player.html에 붙여넣기
