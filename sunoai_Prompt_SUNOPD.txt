<h2>📄 내 저장소 텍스트 파일 목록</h2>
<div class="container">
    <ul id="file-list">
        <li class="loading">파일 목록을 불러오는 중...</li>
    </ul>

    <div id="content-area">파일을 선택하면 내용이 여기에 표시됩니다.</div>
</div>

<script>
    // ⚠️ 수정 필요: 본인의 GitHub 계정명과 저장소(Repo) 이름을 입력하세요.
    const USERNAME = 'gongbyoung'; 
    const REPO = 'GongB';
    const PATH = ''; // 특정 폴더 안의 파일을 보고 싶다면 'folder_name' 입력

    const API_URL = `https://api.github.com/repos/${USERNAME}/${REPO}/contents/${PATH}`;

    async function fetchFiles() {
        try {
            const response = await fetch(API_URL);
            const files = await response.json();
            const listElement = document.getElementById('file-list');
            listElement.innerHTML = '';

            // .txt 파일만 필터링해서 목록 생성
            files.filter(file => file.name.endsWith('.txt')).forEach(file => {
                const li = document.createElement('li');
                li.textContent = file.name;
                li.onclick = () => loadFile(file.download_url);
                listElement.appendChild(li);
            });

            if (listElement.innerHTML === '') {
                listElement.innerHTML = '<li>텍스트 파일이 없습니다.</li>';
            }
        } catch (error) {
            console.error('목록 로드 실패:', error);
            document.getElementById('file-list').innerHTML = '<li>목록을 불러오지 못했습니다.</li>';
        }
    }

    async function loadFile(url) {
        const contentArea = document.getElementById('content-area');
        contentArea.textContent = '불러오는 중...';
        try {
            const response = await fetch(url);
            const text = await response.text();
            contentArea.textContent = text;
        } catch (error) {
            contentArea.textContent = '파일 내용을 불러오는 데 실패했습니다.';
        }
    }

    fetchFiles();
</script>
